## COMMENTS
##
## Contract 1 represents the state of the Door, contract 2 represents the state of the Train
##
## Variable Definitions:
## train_enters - train entering the station
## train_exits - train exiting the station
## train_moving - train moving 
## train_in_station - the train is in a station
## door_opening - the doors of the train are opening
## door_open - the doors of the train are open

CONTRACT:
	NAME:
		train
	VARIABLES:
		train_enters := TRUE
		train_exits := FALSE
		train_moving := TRUE
		train_in_station := FALSE
		door_opening := FALSE
		door_open := FALSE
	ASSUMPTIONS:
		TRUE
	GUARANTEES:
		G(train_enters -> X door_opening)
		G(door_open -> !train_moving)
		G(train_moving -> !door_opening)
		G(door_opening -> X door_open)
		G(train_enters -> train_moving)
		G(train_exits -> train_moving)
		G(train_enters -> X train_in_station)
		G(train_exits -> X !train_in_station)
		G(train_in_station -> !train_moving)
		G(!train_in_station -> train_moving)
		G((train_in_station & !train_exits) -> X train_in_station)

CHECKS:
	COMPATIBILITY_COMP(train)
	CONSISTENCY_COMP(train)
