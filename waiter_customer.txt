## COMMENTS
##
## Contract 1 represents the Waiter, contract 2 represents the Customer
##
## Variable Definitions:
## request - the customer has requested service
## service - the waiter is servicing the customer

CONTRACT:
	NAME:
		waiter1
	VARIABLES:
		request := FALSE
		service := FALSE
	ASSUMPTIONS:
		TRUE
	GUARANTEES:
		G(!request -> X !service)
		G(request -> X service)

CONTRACT:
	NAME:
		waiter2
	VARIABLES:
		request := FALSE
		service := FALSE
	ASSUMPTIONS:
		TRUE
	GUARANTEES:
		G(!request -> X !service)
		G(request -> (X service) | (X X service))

CONTRACT:
	NAME: 
		customer
	VARIABLES:
		request := FALSE
		service := FALSE
	ASSUMPTIONS:
		TRUE
	GUARANTEES:
		(F request)
		G((request & !service) -> X request)
		G(service -> X !request)

CHECKS:
	COMPATIBILITY_COMP(waiter2, customer)
	CONSISTENCY_COMP(waiter2, customer)
	REFINEMENT(waiter1,waiter2)
